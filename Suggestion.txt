def winner_share(gold: int, sailors: int) -> int:

    if sailors == 0:
        return gold
    elif gold % (sailors + 2) == 0:
        return gold/(sailors + 2) * 2
    elif gold % sailors == 0:
        return gold/sailors
    else:
        winner_share(gold, sailors - 1)
